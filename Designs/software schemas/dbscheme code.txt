//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table users as U {
  id guid [pk, increment] // auto-increment
  username varchar
  fname varchar
  lname varchar
  email varchar
  password varchar
  school varchar
  avatarimg varchar
  created_at timestamp
}

Table userfriends as UF {
  id Guid [pk]
  userid guid
  friendid guid
  confirmed bool
 }
 
 Table gamemodes as GM {
  id Guid [pk]
  gamename varchar
  created_at timestamp
 }
 
 Table scoreboards as SB {
  id Guid [pk]
  userid guid
  score int
  maxscore int
  gamemodeid guid
  created_at timestamp
 }
 
 Table customactions as CA {
   id Guid [pk]
   action_group_name varchar
   action varchar
   userid guid
   created_at timestamp
   
 }

// Creating references
// You can also define relaionship separately
// > many-to-one; < one-to-many; - one-to-one
Ref: U.id < CA.userid
Ref: U.id < UF.userid
Ref: U.id < SB.userid
